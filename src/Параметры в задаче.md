```cpp
#include <Arduino.h>
#include <STM32FreeRTOS.h>
HardwareSerial uartPort(PA10, PA9);

struct Param {
    // const char* text;   // либо так 
    String text;           // либо так
    uint32_t delayMs;
};

static Param paramTask1;
static Param paramTask2;

void vTaskFunction( void *pvParameters ) {
    Param* param = (Param*) pvParameters;
    while (1) {
        uartPort.println(param->text);
        vTaskDelay(pdMS_TO_TICKS(param->delayMs));
    }
}

void setup(){
    uartPort.begin(115200);

    paramTask1.delayMs = 500;
    paramTask1.text = "task1";

    paramTask2.delayMs = 700;
    paramTask2.text = "task2";

    xTaskCreate(vTaskFunction, "task1", 1024, (void*)&paramTask1, 1, NULL);
    xTaskCreate(vTaskFunction, "task2", 1024, (void*)&paramTask2, 1, NULL);

    vTaskStartScheduler();
}

void loop(){}
```

Присваивание их неконстантному указателю char* небезопасно, так как теоретически позволяет изменять строковую константу
